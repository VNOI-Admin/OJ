function initAttachmentRow(row) { var $nameField = row.find('input[id$="-display_name"]'); row.find('select[id$="-file"]').on('select2:select', function(e) { $nameField.val(e.params.data.text); }); } $('form').on('submit.attachments', function() { $('#attachment_form_set tr.attachment-row').each(function() { var row = $(this); if (!row.find('select[id$="-file"]').val()) { row.find('input[id$="-display_name"]').val(''); } }); });
